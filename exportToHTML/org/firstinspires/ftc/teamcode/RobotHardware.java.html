97       public void initializeIMU() {
98           //------------------------------------------------------------
99           // IMU - BNO055
100          // Set up the parameters with which we will use our IMU.
101          // + 9 degrees of freedom
102          // + use of calibration file (see calibration program)
103          //------------------------------------------------------------
104  
105          BNO055IMU.Parameters parameters = new BNO055IMU.Parameters();
106          parameters.mode                = BNO055IMU.SensorMode.IMU;
107          parameters.angleUnit           = BNO055IMU.AngleUnit.DEGREES;
108          parameters.accelUnit           = BNO055IMU.AccelUnit.METERS_PERSEC_PERSEC;
109          parameters.calibrationDataFile = "AdafruitImuCalibration.json"; // see the calibration sample opmode
110          parameters.loggingEnabled      = false;
111          //parameters.loggingTag          = "IMU";
112          //parameters.mode                = BNO055IMU.SensorMode.NDOF;
113  
114          parameters.accelerationIntegrationAlgorithm = null;
115  
116          // Retrieve and initialize the IMU. We expect the IMU to be attached to an I2C port
117          // on a Core Device Interface Module, configured to be a sensor of type "AdaFruit IMU",
118          // and named "imu".
119          imu = hardwareMap.get(BNO055IMU.class, "imu");
120          imu.initialize(parameters);
121      }

123      public void startMove(double drive, double strafe, double turn, double scale) {
124          double powerFL = (drive + strafe + turn) * scale;
125          double powerFR = (drive - strafe - turn) * scale;
126          double powerBL = (drive - strafe + turn) * scale;
127          double powerBR = (drive + strafe - turn) * scale;
128  
129          double maxPower = Math.max(Math.max(Math.abs(powerFL), Math.abs(powerFR)), Math.max(Math.abs(powerBL), Math.abs(powerBR))); // ?? Why are we using MAX function here.
130          double max = (maxPower < 1) ? 1 : maxPower;
131  
132          motorFL.setPower(Range.clip(powerFL / max, -1, 1));  // What does this clipping doing?
133          motorFR.setPower(Range.clip(powerFR / max, -1, 1));
134          motorBL.setPower(Range.clip(powerBL / max, -1, 1));
135          motorBR.setPower(Range.clip(powerBR / max, -1, 1));
136      }

146          if (isDown) {
147              servoRClamp.setPosition(RIGHT_FLAP_DOWN_POS);

149          } else {

151              servoLClamp.setPosition(LEFT_FLAP_UP_POS);

153      }
